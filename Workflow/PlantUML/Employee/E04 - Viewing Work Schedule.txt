@startuml
title Melihat Jadwal Kerja

start
: Pengguna memilih "Melihat Jadwal Kerja";
if (Pilihan Pengguna adalah "Melihat Jadwal Kerja") then (Ya)
  : Aplikasi mengambil jadwal kerja pengguna;
  if (Jadwal Kerja Tersedia?) then (Ya)
    : Aplikasi menampilkan jadwal kerja pengguna;
    while (Entri Jadwal Kerja Tersedia?) is (Ya)
      : Untuk setiap entri jadwal kerja dengan atribut-atribut berikut:;
      : Tanggal Kerja (Tanggal);
      : Waktu Masuk (HH:mm AM/PM);
      : Waktu Keluar (HH:mm AM/PM);
      : Total Jam Kerja (HH:mm);
      : Tugas/Proyek (Nama/ID);
      : Lokasi (Kantor/Jarak Jauh);
      : Istirahat yang Diambil (Durasi);

      if (Pengguna memilih entri jadwal kerja) then (Ya)
        : Pengguna mengklik entri jadwal kerja tertentu;
        : Aplikasi menampilkan informasi terperinci;
        : Tanggal Kerja Terperinci (Tanggal);
        : Waktu Masuk Terperinci (HH:mm AM/PM);
        : Waktu Keluar Terperinci (HH:mm AM/PM);
        : Total Jam Kerja Terperinci (HH:mm);
        : Tugas/Proyek Terperinci (Nama/ID);
        : Lokasi Terperinci (Kantor/Jarak Jauh);
        : Istirahat yang Diambil Terperinci (Durasi);
      endif
    endwhile
  else (Tidak)
    : Tidak ada jadwal kerja yang tersedia untuk pengguna;
  endif
else (Tidak)
  : Pengguna memilih opsi lain;
endif

stop
@enduml
