@startuml
title Pengingat Tugas

start
:Pengguna membuka aplikasi Pengingat Tugas;

if (Tugas dengan tenggat waktu?) then (ya)
  :Aplikasi secara otomatis memberikan pengingat untuk tugas-tugas yang memiliki tenggat waktu;
  :Pengguna menerima pemberitahuan tentang tugas-tugas yang harus diselesaikan;

  while (Tugas tersedia dengan detail?) is (ya)
    :Untuk setiap tugas dengan atribut:;
    :ID Tugas;
    :Nama Tugas;
    :Deskripsi;
    :Tenggat Waktu (Tanggal dan Waktu);
    :Prioritas (Rendah, Sedang, Tinggi);
    :Status (Belum Selesai, Selesai);

    if (Pengguna memilih untuk melihat detail tugas) then (ya)
      :Pengguna memilih tugas untuk dilihat detail;
      :Aplikasi menampilkan informasi detail tugas;
    endif

    if (Pengguna memilih untuk menandai tugas sebagai selesai) then (ya)
      :Pengguna memilih tugas untuk ditandai sebagai selesai;
      :Tandai tugas yang dipilih sebagai selesai;
      :Perbarui status tugas menjadi "Selesai";
      :Aplikasi memperbarui status tugas;
      :Aplikasi memberi tahu pengguna tentang penyelesaian tugas;
    else (tidak)
      :Pengguna memilih untuk menunda tugas;
      :Pengguna memilih tugas untuk ditunda;
      :Pengguna menentukan tenggat waktu baru untuk tugas yang dipilih;
      :Perbarui tenggat waktu tugas dengan tanggal baru;
      :Aplikasi memperbarui tenggat waktu tugas;
      :Aplikasi memberi tahu pengguna tentang tenggat waktu baru;
    endif

  endwhile

else (tidak)
  :Tidak ada tugas dengan tenggat waktu yang tersedia;
endif

:Pengguna dapat memilih untuk menambahkan tugas baru dengan detail;
:Pengguna dapat mengatur atribut tugas:;

:Nama Tugas;
:Deskripsi;
:Tenggat Waktu (Tanggal dan Waktu);
:Prioritas (Rendah, Sedang, Tinggi);
:Status (Belum Selesai, Selesai);

if (Pengguna memilih untuk menghapus tugas) then (ya)
  :Pengguna memilih tugas untuk dihapus;
  :Aplikasi menghapus tugas yang dipilih;
  :Aplikasi memberi tahu pengguna tentang penghapusan tugas;
endif

if (Pengguna memilih untuk mengedit tugas) then (ya)
  :Pengguna memilih tugas untuk diedit;
  :Pengguna dapat mengubah detail tugas (Nama, Deskripsi, Tenggat Waktu, Prioritas, Status);
  :Aplikasi memperbarui tugas dengan detail yang diubah;
  :Aplikasi memberi tahu pengguna tentang perubahan tugas;
endif

:Pengguna dapat mengatur atribut tugas:;

:Nama Tugas;
:Deskripsi;
:Tenggat Waktu (Tanggal dan Waktu);
:Prioritas (Rendah, Sedang, Tinggi);
:Status (Belum Selesai, Selesai);

stop
@enduml
