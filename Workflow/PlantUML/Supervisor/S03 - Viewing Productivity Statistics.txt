@startuml

title Viewing Productivity Statistics

start
:Supervisor selects "View Productivity Statistics";
if (Supervisor selects "View Productivity Statistics") then (yes)
  :Application retrieves productivity statistics for the team or department;
  if (Application successfully retrieves statistics) then (yes)
    :Application calculates and displays the following productivity statistics:
    - Average working hours per team member;
    - Total completed tasks;
    - Completed tasks per team member;
    - Project completion rates;
    - Efficiency metrics;
    - Any other relevant productivity indicators;
  else (no)
    :Display error message "Failed to retrieve productivity statistics. Please try again later.";
  endif
else (no)
  :Supervisor selects another option;
endif

stop

@enduml
